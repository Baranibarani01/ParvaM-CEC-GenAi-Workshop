# Program to check palindrome numbers

num = int(input("Enter a number: "))

# Store original number
original = num

# Reverse the number
reverse = 0

while num > 0:
    digit = num % 10
    reverse = reverse * 10 + digit
    num = num // 10

# Check palindrome
if original == reverse:
    print("Palindrome number")
else:
    print("Not a palindrome number")

    # Program to check palindrome (number or string)

def is_palindrome(value):
    # Convert to string for uniform checking
    value_str = str(value)
    
    # Reverse the value
    reverse = value_str[::-1]
    
    # Check palindrome
    if value_str == reverse:
        return True
    else:
        return False


# Few-shot examples
examples = ["121", "1234", "MALAYALAM", "HINDI"]

for item in examples:
    if is_palindrome(item):
        print(f"{item} - is a palindrome")
    else:
        print(f"{item} - is not a palindrome")

🧑‍💼 As a skeptical CEO (security-focused)

Cloud computing means storing data and running applications on remote servers managed by third-party providers instead of on-premise infrastructure.

From a security standpoint:

You’re trusting external vendors with sensitive data
Risks include data breaches, misconfigurations, and compliance issues
However, major providers offer strong encryption, access control, monitoring, and certifications

Bottom line: it’s not inherently insecure—but it shifts responsibility. You gain scalability and cost efficiency, but must enforce strict governance, auditing, and vendor risk management.
